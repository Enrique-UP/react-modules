Javascript

function myFun(name){
console.log(name);
}
myFun("DemoName");



const myFun = (name) => {
console.log(name);
}
myFun("DemoName");



const myFun = name => {
console.log(name);
}
myFun("DemoName");



const myFun = (name, age) => {
console.log(name, age);
}
myFun("DemoName3, 18");
