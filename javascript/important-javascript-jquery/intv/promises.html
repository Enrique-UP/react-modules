Promise is something that is achieved or completed in the future.

वादा वह चीज़ है जो भविष्य में हासिल की जाती है या पूरी की जाती है। 
var promise = new Promise(function(resolve, reject) { 
  const x = 5; 
  const y = 5;
  if(x === y) { 
    resolve(); 
  } else { 
    reject(); 
  } 
});
