var a = "Hello" + 1 + 2 + 3; // for calculation other numbers this is the work of rest operator.

const numbers = ["Hello", 1, 2, 3, 4, 5];
const [one, two, ...rest] = numbers;

document.write(`a= ${a} 
`);
document.write(`${one} 
`);
document.write(`${two} 
`);
document.write(`${rest} 
`);

let sum = 0;
for(let i in rest){
    sum += rest[i];
}
document.write(`Total:- ${sum}
`);
