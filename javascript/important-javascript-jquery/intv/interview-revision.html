// Pure function
function pureFunc(pure){

// Curring function
function currFunc(a,b,c){
    return a*b*c;
}
document.write(currFunc(2,3,4));

// Closure function
function closureFunc(){
    var a = 1;
    function childFunc(){
        return a;
    }
    return childFunc;
}
var getClosureFunc = closureFunc();  
document.write(getClosureFunc());
